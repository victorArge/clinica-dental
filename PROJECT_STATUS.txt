                            COMANDOS DE DESPLIEGUE
================================================================================

# Modo distribuido (MongoDB 3 nodos + Backend 3 nodos + Frontend + Nginx)
docker-compose --profile distributed up -d

# Ver estado de servicios
docker ps --filter "name=clinica"

# Ver logs de un servicio
docker logs -f clinica_node1
docker logs -f clinica_mongo1

# Verificar replica set
docker exec clinica_mongo1 mongosh --quiet --eval "rs.status().members"

# Detener todos los servicios
docker-compose --profile distributed down

# Reconstruir y desplegar
docker-compose --profile distributed up -d --build --force-recreate
